🔧 Firebase Setup Guide

⚠️ You must apply all three rule sets below in your Firebase Console before entries and images will save correctly. This is a one-time setup that takes about 3 minutes.
Step 1 Firestore Database Security Rules
1. Open Firebase Console → Firestore Database → Rules tab
2. Select all existing text and delete it. Paste the code below exactly, then click Publish. 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── USER DOCUMENTS ──────────────────────────────────────
    // Each user can only read and write their own user document.
    match /users/{userId} {
      allow read, write: if request.auth != null
                         && request.auth.uid == userId;
    }

    // ── JOURNAL ENTRIES ─────────────────────────────────────
    // Authenticated users can only access entries they own.
    // IMPORTANT: 'create' uses request.resource (the NEW doc).
    //            'read/update/delete' use resource (existing doc).
    match /entries/{entryId} {
      allow read:   if request.auth != null
                    && resource.data.userId == request.auth.uid;

      allow create: if request.auth != null
                    && request.resource.data.userId == request.auth.uid;

      allow update: if request.auth != null
                    && resource.data.userId == request.auth.uid;

      allow delete: if request.auth != null
                    && resource.data.userId == request.auth.uid;
    }

    // ── MISTAKE TYPES ────────────────────────────────────────
    // Any signed-in user can read mistake types.
    // Only the admin email can create, update, or delete them.
    match /mistakeTypes/{typeId} {
      allow read:  if request.auth != null;
      allow write: if request.auth != null
                   && request.auth.token.email == 'chungzhikai@gmail.com';
    }
  }
}
Step 2 Firebase Storage Security Rules
1. Open Firebase Console → Storage → Rules tab
2. Select all existing text and delete it. Paste the code below exactly, then click Publish. 
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // Images are stored at the path:
    //   journal_images/{userId}/{filename}
    //
    // Each authenticated user can only read and write
    // files inside their own userId folder.
    // Write is further restricted to image files under 10 MB.

    match /journal_images/{userId}/{allPaths=**} {
      allow read:  if request.auth != null
                   && request.auth.uid == userId;

      allow write: if request.auth != null
                   && request.auth.uid == userId
                   && request.resource.size < 10 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }
  }
}
Step 3 Enable Authentication Providers
1. Open Firebase Console → Authentication → Sign-in method
2. Enable the Google provider and the Email/Password provider.
3. Under Authorized domains on the same page, make sure the domain where you serve this HTML file is listed. For local testing, add localhost.
